    function escapeHtml(unsafe) {
        return unsafe.replace(/&/g, '&amp;').replace(/>/g, '&gt;').replace(/</g, '&lt;').replace(/"/g, '&quot;').replace(/'/g, '&apos;');

    function encodePythonSource(){
        var pythonSource = document.getElementById("pythonSource").value;
        var encodedSource = encodeURIComponent(pythonSource);
        var buttonCode = '<br>\n<FORM METHOD="LINK" ACTION="python://'+encodedSource+'"><INPUT TYPE="submit" VALUE="Run With Qpython"></FORM>';
